class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        n = len(s)
        if n == 0:
            return 0

        max_len = float('-inf')
        left = 0
        track = set()


        for right in range(n):

            while s[right] in track:
                track.remove(s[left])
                left +=1

            track.add(s[right])

            max_len = max(max_len, right-left+1)

        return max_len

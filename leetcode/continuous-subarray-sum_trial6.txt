class Solution:
    def checkSubarraySum(self, nums: List[int], k: int) -> bool:
        n = len(nums)
        prefixSum = 0
        remainder_map = {0:-1}

        for i in range(n):
            prefixSum += nums[i]

            rem = prefixSum % k

            if rem in remainder_map:
                if i - remainder_map[rem] >= 2:
                    return True
            else:
                remainder_map[rem] = i

        return False

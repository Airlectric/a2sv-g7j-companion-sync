class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        hold = 0
        seek = 0

        while seek < len(nums):
            if nums[hold] != nums[seek]:
                hold += 1
                nums[hold], nums[seek] = nums[seek], nums[hold]
                
            seek += 1
        
        return hold + 1

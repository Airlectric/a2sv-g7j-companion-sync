class Solution:
    def countPairs(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        seek = 0
        hold = 0
        count = 0

        for i in range(n):
            for j in range(i+1, n):
                if nums[i] + nums[j] < target:
                    count += 1
        
        return count

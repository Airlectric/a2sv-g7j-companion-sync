class Solution:
    def judgeSquareSum(self, c: int) -> bool:
        a = 0
        b = 1 if c== 1 else int(c ** 0.5)


        while a <= b:
            res = a**2 + b **2

            if res < c:
                a += 1
            elif res > c:
                b -= 1
            else:
                return True
        
        return False

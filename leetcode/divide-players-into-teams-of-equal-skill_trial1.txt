class Solution:
    def dividePlayers(self, skill: List[int]) -> int:
        skill.sort()
        left = 0
        right = len(skill)-1

        expected_sum = skill[left] + skill[right]
        res = 0

        while left < right:
            sum_ = skill[left] + skill[right]
            if sum_ == expected_sum:
                res += skill[left] * skill[right]
            else:
                return -1
            
            left += 1
            right -= 1
        
        return res

class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        max_sum = float('-inf')
        left = 0
        track = defaultdict(int)
        n = len(nums)
        curr_sum = 0

        for right in range(n):
            while nums[right] in track:
                track[nums[left]] -= 1
                curr_sum -= nums[left]
                if track[nums[left]] == 0:
                    del track[nums[left]]
                left += 1
            
            track[nums[right]] += 1
            curr_sum += nums[right]
            
            max_sum = max(curr_sum, max_sum)

        return max_sum
